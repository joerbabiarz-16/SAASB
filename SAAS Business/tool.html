import React, { useState, useRef, useEffect } from 'react';

import { Send, Loader, Download, RefreshCw } from 'lucide-react';


const InterviewPrepTool = () => {

  const [messages, setMessages] = useState([]);

  const [input, setInput] = useState('');

  const [isLoading, setIsLoading] = useState(false);

  const [stage, setStage] = useState('welcome');

  const [userData, setUserData] = useState({});

  const [narrative, setNarrative] = useState(null);

  const messagesEndRef = useRef(null);


  const scrollToBottom = () => {

    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });

  };


  useEffect(() => {

    scrollToBottom();

  }, [messages]);


  useEffect(() => {

    // Welcome message

    setMessages([{

      type: 'ai',

      content: `Welcome. I'm here to help you craft a career narrative that's both authentic and strategically positioned.


Most interview prep focuses on rehearsing answers. We're going to do something different—we're going to identify the intentional arc in your story that shows hiring managers who you really are and what you're capable of.


This will take about 15-20 minutes. I'll ask you questions about your background, your pivots, what drives you, and what you're interviewing for.


**Let's start simple: What's your name, and what role are you preparing to interview for?**`

    }]);

  }, []);


  const questionsFlow = {

    welcome: {

      next: 'background',

      prompt: (data) => `Thanks ${data.name}. Now let's understand your story.


**Tell me about your career journey so far.** Where did you start, what pivots have you made, and what brought you to where you are today? Don't worry about being polished—just give me the real story.`

    },

    background: {

      next: 'suffering',

      prompt: () => `Great. Now here's a question that might seem unusual, but it's critical:


**What are you willing to suffer for professionally?** 


Not what sounds good in an interview, but what are you actually willing to struggle through? Long hours? Difficult conversations? Travel? Rejection? Learning curves? What discomfort are you willing to endure to succeed in this role?`

    },

    suffering: {

      next: 'influences',

      prompt: () => `That's honest and helpful. Now let me understand what shapes your thinking.


**What books, people, or experiences have most influenced how you approach work?** Who do you admire professionally and why?`

    },

    influences: {

      next: 'pivots',

      prompt: () => `Good. Now let's look at the pivots in your career.


**Tell me about a time you made a major career change or took on something you weren't "qualified" for on paper.** What was the decision-making process? What did you learn?`

    },

    pivots: {

      next: 'strengths',

      prompt: () => `Now I want to understand your strengths as you see them.


**What do you do better than most people in your field?** Are you a relationship builder? Problem solver? Technical expert? Quick learner? What's your unfair advantage?`

    },

    strengths: {

      next: 'target',

      prompt: (data) => `Excellent. Now let's focus on the specific opportunity.


**Tell me about the ${data.targetRole} role you're interviewing for.** What company? What do you know about what they're looking for? Why do you want this specific role?`

    },

    target: {

      next: 'challenge',

      prompt: () => `Almost there. One more important question:


**What's your biggest concern about this interview process?** What part of your background do you worry might be seen as a weakness or gap? What keeps you up at night about this opportunity?`

    },

    challenge: {

      next: 'generate',

      prompt: () => null

    }

  };


  const handleSend = async () => {

    if (!input.trim() || isLoading) return;


    const userMessage = input.trim();

    setInput('');

    

    setMessages(prev => [...prev, { type: 'user', content: userMessage }]);

    setIsLoading(true);


    // Extract data from user response based on stage

    let newData = { ...userData };

    

    if (stage === 'welcome') {

      // Extract name and role

      newData.rawIntro = userMessage;

      newData.name = userMessage.split(/[,\n]/)[0].replace(/my name is|i'm|i am|hi|hello/gi, '').trim();

      newData.targetRole = userMessage.toLowerCase().includes('role') 

        ? userMessage.split(/role|position/i)[1]?.split(/[,\n.]/)[0].trim() 

        : 'the role you mentioned';

    } else if (stage === 'background') {

      newData.background = userMessage;

    } else if (stage === 'suffering') {

      newData.suffering = userMessage;

    } else if (stage === 'influences') {

      newData.influences = userMessage;

    } else if (stage === 'pivots') {

      newData.pivots = userMessage;

    } else if (stage === 'strengths') {

      newData.strengths = userMessage;

    } else if (stage === 'target') {

      newData.targetDetails = userMessage;

    } else if (stage === 'challenge') {

      newData.challenge = userMessage;

    }


    setUserData(newData);


    // Move to next stage

    const nextStage = questionsFlow[stage].next;

    

    if (nextStage === 'generate') {

      // Generate the narrative

      await generateNarrative(newData);

    } else {

      setStage(nextStage);

      const nextPrompt = questionsFlow[stage].prompt(newData);

      

      setTimeout(() => {

        setMessages(prev => [...prev, { type: 'ai', content: nextPrompt }]);

        setIsLoading(false);

      }, 800);

    }

  };


  const generateNarrative = async (data) => {

    setMessages(prev => [...prev, { 

      type: 'ai', 

      content: '✨ **Analyzing your story and generating your career narrative...**\n\nThis will take about 30-45 seconds. I\'m identifying the intentional arc in your experience and crafting interview responses that position you strategically.' 

    }]);


    try {

      const response = await fetch('https://api.anthropic.com/v1/messages', {

        method: 'POST',

        headers: {

          'Content-Type': 'application/json',

        },

        body: JSON.stringify({

          model: 'claude-sonnet-4-20250514',

          max_tokens: 4000,

          messages: [{

            role: 'user',

            content: `You are a career narrative architect. Your job is to help someone prepare for an interview by identifying the intentional arc in their story and crafting compelling, authentic responses.


PHILOSOPHY TO APPLY:

- Intention: Every career move should show deliberate choice, not randomness

- Suffering: What someone is willing to struggle through reveals authentic motivation

- Punch Card Thinking: Major decisions should show careful consideration (Warren Buffett principle)

- Change as Strength: Successful people handle change successfully and consistently

- Game Theory: Understanding what "game" they're playing and the rules that govern it


USER'S INFORMATION:

Name: ${data.name}

Target Role: ${data.targetRole}

Background: ${data.background}

What They'll Suffer For: ${data.suffering}

Influences: ${data.influences}

Career Pivots: ${data.pivots}

Strengths: ${data.strengths}

Target Role Details: ${data.targetDetails}

Biggest Concern: ${data.challenge}


TASK:

Generate a comprehensive interview preparation document with:


1. NARRATIVE ARC (2-3 paragraphs)

   - Identify the intentional through-line in their story

   - Show how apparent gaps or pivots are actually proof of strategic thinking

   - Frame their background as uniquely valuable for this role


2. CORE INTERVIEW RESPONSES (polished answers to these questions):

   - "Tell me about yourself" (2-minute version that shows the arc)

   - "Why are you interested in this role?"

   - "What's your greatest strength?"

   - "Tell me about a time you faced a significant challenge"

   - "Why should we hire you?"


3. REFRAMING CONCERNS

   - Address their biggest concern head-on

   - Turn it from liability into asset


4. STRATEGIC TALKING POINTS

   - 5-7 key themes they should weave throughout the interview

   - These should feel authentic but be strategically chosen


Make this feel like them, not like a template. Use their actual language and examples. Be specific, not generic.


Format your response in clear markdown with headers.`

          }],

        })

      });


      const responseData = await response.json();

      const narrativeContent = responseData.content

        .filter(item => item.type === 'text')

        .map(item => item.text)

        .join('\n');


      setNarrative(narrativeContent);

      setMessages(prev => [...prev, { 

        type: 'ai', 

        content: `✅ **Your Career Narrative is Ready!**\n\n${narrativeContent}\n\n---\n\n**What's Next?**\n- Review and internalize these themes\n- Practice saying these out loud (not memorizing, but feeling comfortable)\n- Remember: authenticity beats polish every time\n- Use the download button to save this for reference` 

      }]);

      setIsLoading(false);

    } catch (error) {

      setMessages(prev => [...prev, { 

        type: 'ai', 

        content: `I encountered an error generating your narrative. Please try again or contact support if this persists.\n\nError: ${error.message}` 

      }]);

      setIsLoading(false);

    }

  };


  const downloadNarrative = () => {

    if (!narrative) return;

    

    const blob = new Blob([narrative], { type: 'text/markdown' });

    const url = URL.createObjectURL(blob);

    const a = document.createElement('a');

    a.href = url;

    a.download = `${userData.name}_Interview_Narrative.md`;

    document.body.appendChild(a);

    a.click();

    document.body.removeChild(a);

    URL.revokeObjectURL(url);

  };


  const resetTool = () => {

    setMessages([{

      type: 'ai',

      content: `Welcome back. Ready to craft another narrative?\n\n**What's your name, and what role are you preparing to interview for?**`

    }]);

    setStage('welcome');

    setUserData({});

    setNarrative(null);

  };


  return (

    <div className="flex flex-col h-screen bg-gradient-to-br from-purple-50 to-blue-50">

      {/* Header */}

      <div className="bg-white shadow-sm border-b border-gray-200 p-4">

        <div className="max-w-4xl mx-auto flex justify-between items-center">

          <div>

            <h1 className="text-2xl font-bold text-gray-900">Career Narrative Architect</h1>

            <p className="text-sm text-gray-600">AI-Powered Interview Preparation</p>

          </div>

          {narrative && (

            <div className="flex gap-2">

              <button

                onClick={downloadNarrative}

                className="flex items-center gap-2 px-4 py-2 bg-purple-600 text-white rounded-lg hover:bg-purple-700 transition-colors"

              >

                <Download className="w-4 h-4" />

                Download

              </button>

              <button

                onClick={resetTool}

                className="flex items-center gap-2 px-4 py-2 bg-gray-200 text-gray-700 rounded-lg hover:bg-gray-300 transition-colors"

              >

                <RefreshCw className="w-4 h-4" />

                Start Over

              </button>

            </div>

          )}

        </div>

      </div>


      {/* Messages */}

      <div className="flex-1 overflow-y-auto p-4">

        <div className="max-w-4xl mx-auto space-y-4">

          {messages.map((message, index) => (

            <div

              key={index}

              className={`flex ${message.type === 'user' ? 'justify-end' : 'justify-start'}`}

            >

              <div

                className={`max-w-3xl rounded-2xl px-6 py-4 ${

                  message.type === 'user'

                    ? 'bg-purple-600 text-white'

                    : 'bg-white text-gray-900 shadow-md'

                }`}

              >

                <div className="whitespace-pre-wrap prose prose-sm max-w-none">

                  {message.content.split('\n').map((line, i) => {

                    // Handle markdown-style bold

                    if (line.includes('**')) {

                      const parts = line.split(/\*\*(.*?)\*\*/g);

                      return (

                        <p key={i} className="mb-2">

                          {parts.map((part, j) => 

                            j % 2 === 1 ? <strong key={j}>{part}</strong> : part

                          )}

                        </p>

                      );

                    }

                    // Handle headers

                    if (line.startsWith('# ')) {

                      return <h1 key={i} className="text-2xl font-bold mt-4 mb-2">{line.slice(2)}</h1>;

                    }

                    if (line.startsWith('## ')) {

                      return <h2 key={i} className="text-xl font-bold mt-3 mb-2">{line.slice(3)}</h2>;

                    }

                    if (line.startsWith('### ')) {

                      return <h3 key={i} className="text-lg font-bold mt-2 mb-1">{line.slice(4)}</h3>;

                    }

                    // Handle bullets

                    if (line.trim().startsWith('- ')) {

                      return <li key={i} className="ml-4">{line.slice(2)}</li>;

                    }

                    return line ? <p key={i} className="mb-2">{line}</p> : <br key={i} />;

                  })}

                </div>

              </div>

            </div>

          ))}

          {isLoading && (

            <div className="flex justify-start">

              <div className="bg-white rounded-2xl px-6 py-4 shadow-md">

                <Loader className="w-5 h-5 animate-spin text-purple-600" />

              </div>

            </div>

          )}

          <div ref={messagesEndRef} />

        </div>

      </div>


      {/* Input */}

      {!narrative && (

        <div className="bg-white border-t border-gray-200 p-4">

          <div className="max-w-4xl mx-auto">

            <div className="flex gap-2">

              <input

                type="text"

                value={input}

                onChange={(e) => setInput(e.target.value)}

                onKeyPress={(e) => e.key === 'Enter' && handleSend()}

                placeholder="Type your response..."

                className="flex-1 px-4 py-3 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-purple-500"

                disabled={isLoading}

              />

              <button

                onClick={handleSend}

                disabled={isLoading || !input.trim()}

                className="px-6 py-3 bg-purple-600 text-white rounded-lg hover:bg-purple-700 disabled:bg-gray-300 disabled:cursor-not-allowed transition-colors flex items-center gap-2"

              >

                <Send className="w-4 h-4" />

                Send

              </button>

            </div>

            <p className="text-xs text-gray-500 mt-2 text-center">

              Be honest and specific. This is about finding your authentic narrative.

            </p>

          </div>

        </div>

      )}

    </div>

  );

};


export default InterviewPrepTool;
